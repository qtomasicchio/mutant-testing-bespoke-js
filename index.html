  return (interest >= 0.75);
};
      
    
    

    
    
      
    
// Could be mutated as:
const shouldYouTryMutationTesting = (interest) => {
  /* 1 */ return (interest > 0.75);
  /* 2 */ return (interest < 0.75);
  /* 3 */ return false;
  /* 4 */ return true;
};

Theory

    
    
      
  • A mutant should be killed by a unit test
    • 
      
  • A survived mutant can indicated a falsy test
    • 
      
  • Thus, the mutation score is a quality indicator of your tests:
    • 
      
    
# Total detected = # timedOut + # killed
# Total undetected = # survived + # no coverage
# Total mutants = # Total detected + # Total undetected
% Mutation score = # Total detected / # Total mutants * 100
